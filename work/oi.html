function changeTextStyle(element, color, fontWeight, backgroundColor) {
element.style.color = color;
element.style.fontWeight = fontWeight;
element.style.backgroundColor = backgroundColor;
}

function findAndChangeTextStyle() {
var elements = document.querySelectorAll('body *');

elements.forEach(function(element) {
var text = element.textContent.trim();

// Edit colors
switch (text) {
case 'DPOST-PRIOG [C]':
changeTextStyle(element, 'white', 'bold', '#04FF00');
break;
case 'DPOST-PRIOE [D]':
changeTextStyle(element, 'white', 'bold', '#0080FF');
break;
case 'DPOST-TRKD [E]':
changeTextStyle(element, 'white', 'bold', '#FB00FF');
break;
case 'DPOST-PRIOG [F]':
changeTextStyle(element, 'white', 'bold', '#FFAA00');
break;
case 'DPOST-PRIOE [G]':
changeTextStyle(element, 'white', 'bold', '#874888');
break;
default:
break;
}
});
}

window.addEventListener('DOMContentLoaded', findAndChangeTextStyle);
document.addEventListener('DOMSubtreeModified', findAndChangeTextStyle);
